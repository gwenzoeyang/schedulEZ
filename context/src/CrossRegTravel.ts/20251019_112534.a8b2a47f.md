export class CrossRegTravel {
  getArrivalTime(
    origin: ExchangeStop,
    dest: ExchangeStop,
    timeToLeave: string,
  ): string {
    //given the origin and destination, as well as what time the person leaves the origin,
    //return the time at which the person arrives at their destination

    const originTimes = EXCHANGE_BUS_TIMES[origin];
    const idx = originTimes.indexOf(timeToLeave);
    return EXCHANGE_BUS_TIMES[dest][idx];
  }

  getDepartureTime(
    origin: ExchangeStop,
    dest: ExchangeStop,
    timeToArrive: string,
  ): string {
    //given the origin and destination, as well as what time the person gets to the destination,
    //return the time at which the person must leave their origin

    const destTimes = EXCHANGE_BUS_TIMES[dest];
    const idx = destTimes.indexOf(timeToArrive);
    return EXCHANGE_BUS_TIMES[origin][idx];
  }
}
